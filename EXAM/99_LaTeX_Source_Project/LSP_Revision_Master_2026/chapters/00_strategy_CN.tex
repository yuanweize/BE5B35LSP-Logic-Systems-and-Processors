\section{考前须知}

\begin{hack}[考试生存法则]
\textbf{记住三句话：}
\begin{enumerate}
\item \textbf{不会的先跳过}，把会的分拿稳
\item \textbf{写步骤给分}，哪怕算错了过程分也有
\item \textbf{时间不够就蒙}，空着一定0分
\end{enumerate}
\end{hack}

\subsection{考试范围 (老师说的)}

\begin{keybox}[必考 vs 不考]
\textbf{必考：}
\begin{itemize}
\item 有符号/无符号数计算
\item K-Map 4变量化简
\item RS锁存器波形
\item 香农展开
\item 流水线Hazard
\end{itemize}

\textbf{不考：} \sout{Cache计算}、\sout{分支预测器}

\textbf{来源：}老师邮件原话``不考分支预测和Cache，会考流水线''
\end{keybox}

\subsection{答题顺序}

\begin{algorithm}[推荐顺序]
\begin{enumerate}
\item \textbf{数值题} (15min) - 最简单，先拿稳
\item \textbf{K-Map} (20min) - 画图题，细心点
\item \textbf{流水线} (20min) - 套模板
\item \textbf{RS/香农} (25min) - 按步骤来
\item \textbf{检查} (10min) - 检查计算
\end{enumerate}
\end{algorithm}
