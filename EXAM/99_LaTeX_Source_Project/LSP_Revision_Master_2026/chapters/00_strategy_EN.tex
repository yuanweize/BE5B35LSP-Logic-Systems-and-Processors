\section{Before the Exam}

\begin{hack}[Exam Survival Rules]
\textbf{Remember three rules:}
\begin{enumerate}
\item \textbf{Skip what you can't do first} and secure the easy points
\item \textbf{Write the steps for partial credit}; even with a wrong final number, you can still get process points
\item \textbf{If you're running out of time, guess}; leaving it blank is guaranteed 0
\end{enumerate}
\end{hack}

\subsection{Exam Scope (as the instructor said)}

\begin{keybox}[Must-test vs Not tested]
\textbf{Must-test:}
\begin{itemize}
\item Signed/unsigned arithmetic
\item 4-variable K-Map simplification
\item RS latch waveforms
\item Shannon expansion
\item Pipeline hazards
\end{itemize}

\textbf{Not tested:} \sout{Cache calculations}, \sout{branch predictor}

\textbf{Source:} Instructor email: ``Branch prediction and cache are not tested; pipeline will be tested.''
\end{keybox}

\subsection{Recommended Solving Order}

\begin{algorithm}[Suggested order]
\begin{enumerate}
\item \textbf{Number problems} (15min) - easiest, lock points early
\item \textbf{K-Map} (20min) - drawing task, be careful
\item \textbf{Pipeline} (20min) - follow the template
\item \textbf{RS/Shannon} (25min) - do it step by step
\item \textbf{Check} (10min) - verify calculations
\end{enumerate}
\end{algorithm}
