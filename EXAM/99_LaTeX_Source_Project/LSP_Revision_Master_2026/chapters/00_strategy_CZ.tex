\section{Před zkouškou}

\begin{hack}[Pravidla přežití u zkoušky]
\textbf{Pamatuj si tři pravidla:}
\begin{enumerate}
\item \textbf{Co neumíš, nejdřív přeskoč} a zajisti si snadné body
\item \textbf{Piš postup pro dílčí body}; i když se spleteš ve výsledku, body za postup často zůstanou
\item \textbf{Když dochází čas, tipuj}; nechat prázdné je jistá 0
\end{enumerate}
\end{hack}

\subsection{Rozsah zkoušky (podle učitele)}

\begin{keybox}[Povinné vs Nezkoušené]
\textbf{Povinné:}
\begin{itemize}
\item Výpočty se znaménkem / bez znaménka
\item Zjednodušení K-map pro 4 proměnné
\item Průběhy RS záchytu
\item Shannonův rozvoj
\item Hazard v pipeline
\end{itemize}

\textbf{Nezkoušené:} \sout{výpočty cache}, \sout{prediktor skoků}

\textbf{Zdroj:} Email od učitele: ``Predikce skoků a cache se nezkouší; bude pipeline.''
\end{keybox}

\subsection{Doporučené pořadí řešení}

\begin{algorithm}[Doporučené pořadí]
\begin{enumerate}
\item \textbf{Číselné úlohy} (15min) - nejjednodušší, vezmi body hned
\item \textbf{K-Map} (20min) - kreslení, buď pečlivý
\item \textbf{Pipeline} (20min) - podle šablony
\item \textbf{RS/Shannon} (25min) - krok za krokem
\item \textbf{Kontrola} (10min) - ověř výpočty
\end{enumerate}
\end{algorithm}
